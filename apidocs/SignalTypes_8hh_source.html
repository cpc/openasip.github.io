55 /// Signal holds address.
56 ADDRESS,

58 INSTRUCTIONWORD,

60 /// instruction.
61 FETCHBLOCK,

63 READ_REQUEST,
64 /// Signal to make burst read request.
65 BURST_READ_REQUEST,

67 READ_REQUEST_READY,

69 BURST_READ_REQUEST_READY,
70 /// Signal to make write request.
71 WRITE_REQUEST,
72 /// Signal to make either read or write request.
73 READ_WRITE_REQUEST,
