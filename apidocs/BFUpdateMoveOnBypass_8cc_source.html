41bool BFUpdateMoveOnBypass::operator()() {
42
43 auto& dm = dst_.move();
44 auto& sm = src_.move();
45
46 // update the move
47 dm.setSource(sm.source().copy());
48
49 if (!src_.isSourceOperation()) {
50 return true;
51 }
52
53 ProgramOperationPtr srcOp = src_.sourceOperationPtr();
54 srcOp->addOutputNode(dst_);
55 dst_.setSourceOperationPtr(srcOp);
56
57 // set fu annotations
58 for (int j = 0; j < sm.annotationCount(); j++) {
59 TTAProgram::ProgramAnnotation anno = sm.annotation(j);
60 if (anno.id() == ProgramAnnotation::ANN_ALLOWED_UNIT_SRC) {
61 dm.addAnnotation(
62 ProgramAnnotation(
63 ProgramAnnotation::ANN_ALLOWED_UNIT_SRC,
64 anno.payload()));
65 }
66 if (anno.id() ==
67 ProgramAnnotation::ANN_CONN_CANDIDATE_UNIT_SRC) {
68 dm.addAnnotation(
69 ProgramAnnotation(
70 ProgramAnnotation::ANN_CONN_CANDIDATE_UNIT_SRC,
71 anno.payload()));
72 }
73 if (anno.id() ==
74 ProgramAnnotation::ANN_REJECTED_UNIT_SRC) {
75 dm.addAnnotation(
76 ProgramAnnotation(
77 ProgramAnnotation::ANN_REJECTED_UNIT_SRC,
78 anno.payload()));
79 }
80 }
81 return true;
82}

84void BFUpdateMoveOnBypass::undoOnlyMe() {
85 // unset programoperation from bypassed
86 if (dst_.isSourceOperation()) {
87 ProgramOperation& srcOp = dst_.sourceOperation();
88 srcOp.removeOutputNode(dst_);
89 dst_.unsetSourceOperation();
90
91 // unset fu annotations
92 dst_.move().removeAnnotations(
93 TTAProgram::ProgramAnnotation::ANN_ALLOWED_UNIT_SRC);
94 dst_.move().removeAnnotations(
95 TTAProgram::ProgramAnnotation::ANN_CONN_CANDIDATE_UNIT_SRC);
96 dst_.move().removeAnnotations(
97 TTAProgram::ProgramAnnotation::ANN_REJECTED_UNIT_SRC);
98 }
99 // do the actual unmerge by returning source to original register.
100 dst_.move().setSource(src_.move().destination().copy());
101}
