58/**
59 * Executes the "run" command.
60 *
61 * Starts the simulation.
62 *
63 * @param arguments No arguments needed.
64 * @return Always true.
65 * @exception NumberFormatException Is never thrown by this command.
66 *
67 */
68bool
69RunCommand::execute(const std::vector<DataObject>&) {
70 if (!checkSimulationNotAlreadyRunning() ||
71 !checkSimulationInitialized()) {
72
73 return false;
74 }
75
76 simulatorFrontend().run();
77
78 printStopInformation();
79
80 return true;
81}
