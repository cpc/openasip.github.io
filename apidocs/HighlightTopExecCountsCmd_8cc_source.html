56/**
57 * Executes the command.
58 */
59bool
60HighlightTopExecCountsCmd::Do() {
61 ProximDisassemblyWindow* window = ProximToolbox::disassemblyWindow();
62 assert(window != NULL);
63 DisasmTopCountAttrProvider* attrProvider =
64 new DisasmTopCountAttrProvider(*ProximToolbox::frontend(), 10);
65
66 if (ProximToolbox::frontend()->isProgramLoaded()) {
67 attrProvider->update();
68 }
69 window->setMoveAttrProvider(attrProvider);
70 window->Refresh();
71 return true;
72}
