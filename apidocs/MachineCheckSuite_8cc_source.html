70/**
71 * Runs all checks in the check suite.
72 *
73 * @param machine Machine to check.
74 * @param results Result container where error messages are added.
75 * @return True, if all checks passed, false otherwise.
76 */
77bool
78MachineCheckSuite::run(
79 const TTAMachine::Machine& machine,
80 MachineCheckResults& results) {
81
82 bool pass = true;
83 for (int i = 0; i < checkCount(); i++) {
84 if (!checks_[i]->check(machine, results)) {
85 pass = false;
86 }
87 }
88 return pass;
89}
