33CallPathTracker::addProcedureTransfer(
34 ClockCycleCount cycle,
35 InstructionAddress address,
36 InstructionAddress sourceAddress,
37 bool isEntry) {
38
39 if (transfers_.size() > historyLength_)

41 ProcedureTransfer transfer;

44 transfer.sourceAddress = sourceAddress;
45 transfer.isEntry = isEntry;

49/**
50 * Returns the history in a "back trace form" with calls and returns not
51 * contributing to the current call path stripped out.
52 */
53const CallPathTracker::ProcedureTransferQueue
54CallPathTracker::backTrace() const {
55 ProcedureTransferQueue allTransfers = transfers();
56 ProcedureTransferQueue bt;
57 int callsToSkip = 0;
58 CallPathTracker::ProcedureTransferQueue::const_reverse_iterator i =
59 allTransfers.rbegin();
60 // Define end iterator as an explilcit variable to avoid compiling
61 // bug under Mac OS X.
62 CallPathTracker::ProcedureTransferQueue::const_reverse_iterator e =
63 allTransfers.rend();
64 for (; i != e; ++i) {
65 CallPathTracker::ProcedureTransfer tr = *i;
66 if (!tr.isEntry) {
67 ++callsToSkip;
68 continue;
69 }
70 if (callsToSkip == 0) {
71 bt.push_back(tr);
72 } else {
73 callsToSkip--;
74 }
75 }
76 return bt;
77}
