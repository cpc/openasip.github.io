44/**
45 * Edge of the control flow graph (control flow edge). Each edge implies a
46 * possible change of execution flow from the source basic block to the
47 * target basic block.
48 *
49 */
50class ControlFlowEdge : public GraphEdge {
51public:
52 enum CFGEdgePredicate {
53 CFLOW_EDGE_NORMAL,
54 CFLOW_EDGE_TRUE,
55 CFLOW_EDGE_FALSE,

57 CFLOW_EDGE_FAKE
58 };
59 enum CFGEdgeType {
60 CFLOW_EDGE_JUMP,
61 CFLOW_EDGE_CALL,
62 CFLOW_EDGE_FALLTHROUGH
63 };
64
65 ControlFlowEdge(
66 CFGEdgePredicate edgePredicate = CFLOW_EDGE_NORMAL,
67 CFGEdgeType edgeType = CFLOW_EDGE_JUMP);
68
69 virtual ~ControlFlowEdge();
70

74 bool isFalseEdge() const;
75 bool isCallPassEdge() const;
76 bool isJumpEdge() const;
77 bool isFallThroughEdge() const;
78 bool isLoopBreakEdge() const;
79 bool isBackEdge() const;

82
83 CFGEdgePredicate edgePredicate() const { return edgePredicate_; }

93private:

97};
