49/**
50 * Constructor.
51 *
52 * param type The type of the drop down menu.
53 */
54DropDownMenu::DropDownMenu(MenuType type) : wxMenu() {
55
56 OSEdTextGenerator& texts = OSEdTextGenerator::instance();
57
58 switch (type) {
59 case MENU_PATH: {
60 format fmt = texts.text(OSEdTextGenerator::TXT_MENU_ADD_MODULE);
61 Append(
62 OSEdConstants::CMD_ADD_MODULE,
63 WxConversion::toWxString(fmt.str()));
64 }
65 break;
66 case MENU_MODULE: {
67 format fmt = texts.text(OSEdTextGenerator::TXT_MENU_ADD_OPERATION);
68 Append(
69 OSEdConstants::CMD_ADD_OPERATION,
70 WxConversion::toWxString(fmt.str()));
71
72 fmt = texts.text(OSEdTextGenerator::TXT_MENU_REMOVE_MODULE);
73 Append(
74 OSEdConstants::CMD_REMOVE_MODULE,
75 WxConversion::toWxString(fmt.str()));
76
77 fmt = texts.text(OSEdTextGenerator::TXT_MENU_BUILD);
78 Append(
79 OSEdConstants::CMD_BUILD, WxConversion::toWxString(fmt.str()));
80 }
81 break;
82 case MENU_OPERATION: {
83 format fmt = texts.text(OSEdTextGenerator::TXT_MENU_MODIFY_PROPERTIES);
84 Append(
85 OSEdConstants::CMD_PROPERTIES,
86 WxConversion::toWxString(fmt.str()));
87
88 fmt = texts.text(OSEdTextGenerator::TXT_MENU_MODIFY_BEHAVIOR);
89 Append(
90 OSEdConstants::CMD_MODIFY_BEHAVIOR,
91 WxConversion::toWxString(fmt.str()));
92
93 fmt = texts.text(OSEdTextGenerator::TXT_MENU_REMOVE_OPERATION);
94 Append(
95 OSEdConstants::CMD_REMOVE_OPERATION,
96 WxConversion::toWxString(fmt.str()));
97
98 fmt = texts.text(OSEdTextGenerator::TXT_MENU_SIMULATE);
99 Append(
100 OSEdConstants::CMD_SIMULATE,
101 WxConversion::toWxString(fmt.str()));
102 }
103 break;
104 default:
105 break;
106 }
107
108 updateMenu();
109}

127/**
128 * Updates the menus enabled/disabled status.
129 */
130void
131DropDownMenu::updateMenu() {
132 CommandRegistry* registry = wxGetApp().mainFrame()->registry();
133 GUICommand* command = registry->firstCommand();
134 while (command != NULL) {
135 if (FindItem(command->id()) != NULL) {
136 if (command->isEnabled()) {
137 Enable(command->id(), true);
138 } else {
139 Enable(command->id(), false);
140 }
141 }
142 command = registry->nextCommand();
143 }
144}
