114 void validate() {
115 if (outputDirectory.empty()) {
116 outputDirectory = "proge-output";
117 }
118 if (sharedOutputDirectory.empty()) {
119 sharedOutputDirectory = outputDirectory;
120 }
121 FileSystem::expandTilde(sharedOutputDirectory);
122 sharedOutputDirectory =
123 FileSystem::absolutePathOf(sharedOutputDirectory);
124 if (entityName.empty()) {
125 entityName = "tta0";
126 }
127 if (languageStr == "verilog") {
128 language = ProGe::HDL::Verilog;
129 } else {
130 language = ProGe::HDL::VHDL;
131 }
132 if (hdbList.empty()) {
133 hdbList.emplace_back("generate_base32.hdb");
134 hdbList.emplace_back("generate_lsu_32.hdb");
