51void
52OTASimulationController::advanceMachineCycle(unsigned pcAdd) {
53
54 const unsigned int core = 0;
55 MachineState& machineState = *machineStates_[core];
56 //InstructionMemory& imem = *instructionMemories_[core];
57 MemorySystem& memorySystem = frontend_.memorySystem(core);
58 GCUState& gcu = machineState.gcuState();
59
60 machineState.endClockOfAllFUStates();
61
62 if (!gcu.isIdle()) {
63 gcu.endClock();
64 }
65
66 memorySystem.advanceClockOfLocalMemories();
67 memorySystem.advanceClockOfSharedMemories();
68 machineState.advanceClockOfAllFUStates();
69
70 gcu.programCounter() += pcAdd;
71 if (!gcu.isIdle())
72 gcu.advanceClock();
73
74 machineState.advanceClockOfAllGuardStates();
75}

77/**
78 * Simulates an instruction cycle, both its explicit and implicit instructions.
79 */
80bool
81OTASimulationController::simulateCycle() {
82
83 std::vector<InstructionAddress> lastExecutedInstructions =
84 lastExecutedInstruction_;
85
86 bool finished = false;
87
88 int finishedCoreCount = 0;
89
90 const unsigned int core = 0;
91 MachineState& machineState = *machineStates_[core];
92 InstructionMemory& imem = *instructionMemories_[core];
93 GCUState& gcu = machineState.gcuState();
94 const InstructionAddress pc = gcu.programCounter();
95
96 try {
97 machineState.clearBuses();
98
99 bool exitPoint = false;
100 if (imem.hasInstructionAt(pc)) {
101 ExecutableInstruction& instruction = imem.instructionAt(pc);
102 instruction.execute();
103
104 lastExecutedInstructions[core] = pc;
105
106 advanceMachineCycle(4);
107
108 if (imem.hasImplicitInstructionsAt(pc)) {
109 const auto& implInstructions = imem.implicitInstructionsAt(pc);
110 for (size_t i = 0; i < implInstructions.size(); ++i) {
111 ExecutableInstruction& implInstruction =
112 *implInstructions.at(i);
113 implInstruction.execute();
114 exitPoint |= implInstruction.isExitPoint();
115 advanceMachineCycle(0);
116 }
117 }
118
119 exitPoint |= instruction.isExitPoint();
120 }
121
122 // check if the instruction was a return point from the program or
123 // the next executed instruction would be sequentially over the
124 // instruction space (PC+1 would overflow out of the program)
125 if (exitPoint ||
126 gcu.programCounter() == firstIllegalInstructionIndex_) {
127 machineState.setFinished();
128 ++finishedCoreCount;
129 }
130 } catch (const Exception& e) {
131 frontend_.reportSimulatedProgramError(
132 SimulatorFrontend::RES_FATAL,
133 e.errorMessage());
134 prepareToStop(SRE_RUNTIME_ERROR);
135 return false;
136 }
137
138 if (finishedCoreCount > 0)
139 finished = true;
140
141 frontend_.eventHandler().handleEvent(SimulationEventHandler::SE_CYCLE_END);
142
143 lastExecutedInstruction_ = lastExecutedInstructions;
144
145 // this is the instruction count in case of OTA
146 ++clockCount_;
147
148 if (finished) {
149 state_ = STA_FINISHED;
150 stopRequested_ = true;
151 return false;
152 }
153
154 frontend_.eventHandler().handleEvent(
155 SimulationEventHandler::SE_NEW_INSTRUCTION);
156 return true;
157}
