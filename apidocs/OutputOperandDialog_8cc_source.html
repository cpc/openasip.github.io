192/**
193 * Updates the element width choice box list.
194**/
195void
196OutputOperandDialog::updateElementWidths() {
197
198 Operand::OperandType operType = static_cast<Operand::OperandType>(type_);
199
200 if (operType == Operand::RAW_DATA) {
201 // element width for raw data can be arbitrary up to the max width
202 int elemWidth = 1;
203 int lastValidWidth = 1;
204 while (elemCount_*elemWidth <= SIMD_WORD_WIDTH) {
205 lastValidWidth = elemWidth;
206 elemWidth *= 2;
207 }
208
209 // degrade current element width if it is too big
210 if (elemWidth_ > lastValidWidth) {
211 elemWidth_ = lastValidWidth;
212 }
213 elementWidthSpinCtrl_->SetRange(1, lastValidWidth);
214 elementWidthSpinCtrl_->SetValue(elemWidth_);
215 } else if (operType == Operand::SINT_WORD || operType == Operand::UINT_WORD) {
216 // element width for integers is 8 to 32
217 elementWidthSpinCtrl_->SetRange(8, 32);
218 elementWidthSpinCtrl_->SetValue(elemWidth_);
219 } else {
220 // element width for other types is their default type width
221 elementWidthSpinCtrl_->SetRange(elemWidth_, elemWidth_);
222 elementWidthSpinCtrl_->SetValue(elemWidth_);
223 }
224}

226/**
227 * Updates the element count choice box list.
228**/
229void
230OutputOperandDialog::updateElementCounts() {
231
232 elementCountChoice_->Clear();
233
234 // update the list so that only shorter or equal than SIMD_WORD_WIDTH
235 // width*count combinations are listed
236 int elemCount = 1;
237 int elemCountIndex = 0;
238 while (elemCount*elemWidth_ <= SIMD_WORD_WIDTH) {
239 if (elemCount < elemCount_) {
240 ++elemCountIndex;
241 }
242 elementCountChoice_->Append(WxConversion::toWxString(elemCount));
243 elemCount *= 2;
244 }
245 elementCountChoice_->SetSelection(elemCountIndex);
246}

275/**
276 * Handles the event when OK button is pushed.
277 */
278void
279OutputOperandDialog::onOk(wxCommandEvent&) {
280 TransferDataFromWindow();
281 ObjectState* root = new ObjectState("");
282 root->setAttribute(Operand::OPRND_ID, operand_->index());
283
284 int selected = outputTypesComboBox_->GetSelection();
285 Operand::OperandType type = operandTypes_[selected];
286
287 switch(type)
288 {
289 case Operand::SINT_WORD:
290 root->setAttribute(Operand::OPRND_TYPE, Operand::SINT_WORD_STRING);
291 break;
292 case Operand::UINT_WORD:
293 root->setAttribute(Operand::OPRND_TYPE, Operand::UINT_WORD_STRING);
294 break;
295 case Operand::FLOAT_WORD:
296 root->setAttribute(Operand::OPRND_TYPE, Operand::FLOAT_WORD_STRING);
297 break;
298 case Operand::DOUBLE_WORD:
299 root->setAttribute(Operand::OPRND_TYPE, Operand::DOUBLE_WORD_STRING);
300 break;
301 case Operand::HALF_FLOAT_WORD:
302 root->setAttribute(
303 Operand::OPRND_TYPE, Operand::HALF_FLOAT_WORD_STRING);
304 break;
305 case Operand::BOOL:
306 root->setAttribute(Operand::OPRND_TYPE, Operand::BOOL_STRING);
307 break;
308 case Operand::RAW_DATA:
309 root->setAttribute(Operand::OPRND_TYPE, Operand::RAW_DATA_STRING);
310 break;
311 case Operand::SLONG_WORD:
312 root->setAttribute(
313 Operand::OPRND_TYPE, Operand::SLONG_WORD_STRING);
314 break;
315 case Operand::ULONG_WORD:
316 root->setAttribute(
317 Operand::OPRND_TYPE, Operand::ULONG_WORD_STRING);
318 break;
319 default:
320 root->setAttribute(Operand::OPRND_TYPE, Operand::SINT_WORD_STRING);
321 break;
322 }
323
324 root->setAttribute(Operand::OPRND_ELEM_WIDTH, elemWidth_);
325 root->setAttribute(Operand::OPRND_ELEM_COUNT, elemCount_);
326
327 root->setAttribute(Operand::OPRND_MEM_ADDRESS, false);
328 root->setAttribute(Operand::OPRND_MEM_DATA, memData_);
329
330 operand_->loadState(root);
331 delete root;
332 EndModal(wxID_OK);
333}
