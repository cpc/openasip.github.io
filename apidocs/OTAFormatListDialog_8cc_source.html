162/**
163 * Transfers data from the machine to the dialog widgets.
164 */
165bool
166OTAFormatListDialog::TransferDataToWindow() {
167
168 // update OTAFormat list
169 OTAFormatList_->DeleteAllItems();
170 Machine::OperationTriggeredFormatNavigator navigator =
171 machine_->operationTriggeredFormatNavigator();
172 for (int i = 0; i < navigator.count(); i++) {
173 OTAFormatList_->InsertItem(
174 i, WxConversion::toWxString(navigator.item(i)->name()));
175 }
176 updateOperationList();
177 return wxDialog::TransferDataToWindow();
178}

181/**
182 * Updates the OTAFormat Operation list.
183 */
184void
185OTAFormatListDialog::updateOperationList() {
186
187
188 operationList_->DeleteAllItems();
189 operationList_->Enable();
190
191 OperationTriggeredFormat* f = selectedOTAFormat();
192
193 if (f == NULL) {
194 FindWindow(ID_ADD_OPERATION)->Disable();
195 return;
196 }
197
198 for (int i = 0; i < f->operationCount(); i++) {
199 operationList_->InsertItem(
200 0, WxConversion::toWxString(f->operationAtIndex(i)));
201 }
202
203 FindWindow(ID_ADD_OPERATION)->Enable();
204
205 wxListEvent dummy;
206 onOperationSelection(dummy);
207}

281bool
282OTAFormatListDialog::validFormatName() const {
283 if (OTAFormatName_ == RISCVFields::RISCV_R_TYPE_NAME ||
284 OTAFormatName_ == RISCVFields::RISCV_R1_TYPE_NAME ||
285 OTAFormatName_ == RISCVFields::RISCV_R3R_TYPE_NAME ||
286 OTAFormatName_ == RISCVFields::RISCV_R1R_TYPE_NAME ||
287 OTAFormatName_ == RISCVFields::RISCV_I_TYPE_NAME ||
288 OTAFormatName_ == RISCVFields::RISCV_S_TYPE_NAME ||
289 OTAFormatName_ == RISCVFields::RISCV_B_TYPE_NAME ||
290 OTAFormatName_ == RISCVFields::RISCV_U_TYPE_NAME ||
291 OTAFormatName_ == RISCVFields::RISCV_J_TYPE_NAME) {
292 return true;
293 }
294 return false;
295}

298/**
299 * Handles the Add tempalte button event.
300 *
301 * Adds a new OTAFormat to the machine.
302 */
303void
304OTAFormatListDialog::onAddOTAFormat(wxCommandEvent&) {
305 if (!TransferDataFromWindow()) {
306 assert(false);
307 }
308 string trimmedName =
309 WxConversion::toString(OTAFormatName_.Trim(false).Trim(true));
310
311 // Check the name validity.
312 if (!validFormatName()) {
313 std::string message = "Format name is illegal. Legal names are:\n" + \
314 RISCVFields::RISCV_R_TYPE_NAME + "\n" + \
315 RISCVFields::RISCV_R1_TYPE_NAME + "\n" + \
316 RISCVFields::RISCV_R3R_TYPE_NAME + "\n" + \
317 RISCVFields::RISCV_R1R_TYPE_NAME + "\n" + \
318 RISCVFields::RISCV_I_TYPE_NAME + "\n" + \
319 RISCVFields::RISCV_S_TYPE_NAME + "\n" + \
320 RISCVFields::RISCV_B_TYPE_NAME + "\n" + \
321 RISCVFields::RISCV_U_TYPE_NAME + "\n" + \
322 RISCVFields::RISCV_J_TYPE_NAME + "\n";
323 InformationDialog warning(
324 this, WxConversion::toWxString(message));
325 warning.ShowModal();
326 return;
327 }
328
329 Machine::OperationTriggeredFormatNavigator navigator =
330 machine_->operationTriggeredFormatNavigator();
331
332 if (navigator.hasItem(trimmedName)) {
333 ProDeTextGenerator* prodeTexts =
334 ProDeTextGenerator::instance();
335 format message =
336 prodeTexts->text(ProDeTextGenerator::MSG_ERROR_SAME_NAME);
337 format a_tmplate =
338 prodeTexts->text(ProDeTextGenerator::COMP_AN_OTA_FORMAT);
339 format machine =
340 prodeTexts->text(ProDeTextGenerator::COMP_MACHINE);
341 format tmplate =
342 prodeTexts->text(ProDeTextGenerator::COMP_OTA_FORMAT);
343 message % trimmedName % a_tmplate.str() % machine.str() %
344 tmplate.str();
345 WarningDialog warning(this, WxConversion::toWxString(message.str()));
346 warning.ShowModal();
347 return;
348 }
349 new OperationTriggeredFormat(trimmedName, *machine_);
350 OTAFormatName_ = _T("");
351 TransferDataToWindow();
352}

355/**
356 * Enables and disables the Add OTAFormat button when text is entered in the
357 * OTAFormat name widget.
358 */
359void
360OTAFormatListDialog::onOTAFormatName(wxCommandEvent&) {
361 if (!TransferDataFromWindow()) {
362 assert(false);
363 }
364 wxString trimmedName = OTAFormatName_.Trim(false).Trim(true);
365 Machine::OperationTriggeredFormatNavigator nav =
366 machine_->operationTriggeredFormatNavigator();
367 if (trimmedName == _T("")) {
368 FindWindow(ID_ADD_OTA_FORMAT)->Disable();
369 //Only 6 formats in RISC-V + 3 custom formats
370 } else if (nav.count() > 8) {
371 FindWindow(ID_ADD_OTA_FORMAT)->Disable();
372 } else {
373 FindWindow(ID_ADD_OTA_FORMAT)->Enable();
374 }
375}
