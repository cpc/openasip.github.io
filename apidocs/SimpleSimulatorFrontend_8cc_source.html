38SimpleSimulatorFrontend::SimpleSimulatorFrontend(
39 TCEString machineFile, bool useCompiledSimulation) {
40 simFront_ =
41 new SimulatorFrontend(
42 useCompiledSimulation ?
43 SimulatorFrontend::SIM_COMPILED :
44 SimulatorFrontend::SIM_NORMAL);
45 simFront_->setZeroFillMemoriesOnReset(false);
46 try {
47 simFront_->loadMachine(machineFile);
48 } catch (Exception& e) {
49 std::cerr
50 << e.errorMessage() + " in " +
51 e.fileName() + ":" +
52 e.procedureName() + ":"
53 << e.lineNum() << std::endl;
54 abort();
55 }
56}
