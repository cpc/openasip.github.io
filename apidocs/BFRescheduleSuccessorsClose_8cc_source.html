42bool BFRescheduleSuccessorsClose::operator()() {
43
44#ifdef DEBUG_BUBBLFISH_SCHEDULER
45 std::cerr << "\t\t\tRescheduling successors close" << std::endl;
46#endif
47 bool pushed = false;
48 auto succs = ddg().successors(mn_);
49 for (auto i: succs) {
50 if (i->isScheduled()) {
51 if (ddg().earliestCycle(*i) < i->cycle()) {
52 auto resched = new BFRescheduleResultClose(sched_, *i, 0);
53 pushed |= runPreChild(resched);
54 }
55 }
56 }
57 return pushed;
58}
