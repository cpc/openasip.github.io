111/**
112 * Loads the state of the object from the given ObjectState object.
113 *
114 * @param state ObjectState from which the state is loaded.
115 * @exception ObjectStateLoadingException If the given ObjectState instance
116 * is invalid.
117 */
118void
119ProDeOptions::loadState(const ObjectState* state) {
120 deleteAllKeyboardShortcuts();
121 deleteAllToolbarButtons();
122
123 const string procName = "ProDeOptions::loadState";
124
125 GUIOptions::loadState(state);
126
127 try {
128 undoStackSize_ = state->intAttribute(OSKEY_UNDO_STACK_SIZE);
129 } catch (...) {
130 deleteAllKeyboardShortcuts();
131 deleteAllToolbarButtons();
132 throw ObjectStateLoadingException(__FILE__, __LINE__, procName);
133 }
134}

141ObjectState*
142ProDeOptions::saveState() const {
143 ObjectState* state = GUIOptions::saveState();
144 state->setAttribute(OSKEY_UNDO_STACK_SIZE, undoStackSize_);
145 return state;
