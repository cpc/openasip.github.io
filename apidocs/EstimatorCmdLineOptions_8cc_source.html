65EstimatorCmdLineOptions::EstimatorCmdLineOptions() : CmdLineOptions("") {
66 addOption(
67 new StringCmdLineOptionParser(
68 SWL_TPEF,
69 "sets the TTA program exchange format file (TPEF) from which to "
70 "load the estimated program (required for energy estimation only)",
71 SWS_TPEF));
72 addOption(
73 new StringCmdLineOptionParser(
74 SWL_TRACE,
75 "sets the simulation trace database (TraceDB) from which to "
76 "load the simulation data of the estimated program (required for "
77 "energy estimation only)",
78 SWS_TRACE));
79 addOption(
80 new BoolCmdLineOptionParser(
81 SWL_TOTALAREA,
82 "run total area estimation",
83 SWS_TOTALAREA));
84 addOption(
85 new BoolCmdLineOptionParser(
86 SWL_LONGESTPATH,
87 "run longest path estimation",
88 SWS_LONGESTPATH));
89 addOption(
90 new BoolCmdLineOptionParser(
91 SWL_TOTALENERGY,
92 "run total energy consumption estimation",
93 SWS_TOTALENERGY));
94}
