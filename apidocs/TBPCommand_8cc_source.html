59/**
60 * Executes the "tbp" command.
61 *
62 * Sets a temporary breakpoint, which is automatically deleted after
63 * the first time it stops the simulation. The arguments args are the
64 * same as for the bp command.
65 *
66 * @param arguments The address (and condition of the breakpoint).
67 * @return True in case simulation is initialized and arguments are ok.
68 * @exception NumberFormatException Is never thrown by this command.
69 */
70bool
71TBPCommand::execute(const std::vector<DataObject>& arguments) {
72 Breakpoint breakpoint(simulatorFrontend(), 0);
73 if (!parseBreakpoint(arguments, breakpoint)) {
74 return false;
75 }
76 breakpoint.setDeletedAfterTriggered(true);
77
78 StopPointManager& stopPointManager =
79 simulatorFrontend().stopPointManager();
80 return printBreakpointInfo(stopPointManager.add(breakpoint));
81}
