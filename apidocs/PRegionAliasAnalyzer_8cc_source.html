72 return ALIAS_UNKNOWN;

74 int pregion1 = m1.annotation(

76 int pregion2 = m2.annotation(
77 0, ProgramAnnotation::ANN_PARALLEL_REGION_ID).intValue();
78
79 if (pregion1 != pregion2) {
