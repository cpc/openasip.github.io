95/**
96 * Transfers settings data from the simulator frontend to the dialog widgets.
97 */
98bool
99SimulatorSettingsDialog::TransferDataToWindow() {
100
101 execTrace_ = simulator_.executionTracing();
102 busTrace_ = simulator_.busTracing();
103 ptTrace_ = simulator_.procedureTransferTracing();
104 rfTrace_ = simulator_.rfAccessTracing();
105
106 profileDataSaving_ = simulator_.profileDataSaving();
107 utilizationDataSaving_ = simulator_.utilizationDataSaving();
108 nextInstructionPrinting_ = simulator_.nextInstructionPrinting();
109 simulationTimeStatistics_ = simulator_.simulationTimeStatistics();
110
111 fuConflictDetection_ = simulator_.fuResourceConflictDetection();
112
113 historySave_ = lineReader_.saveInputHistoryToFile();
114 historySize_ = lineReader_.inputHistoryMaxLength();
115 historyFile_ = WxConversion::toWxString(
116 lineReader_.inputHistoryFilename());
117
118 return wxDialog::TransferDataToWindow();
119}

122/**
123 * Updates the simulator settings and closes the dialog when the OK-button
124 * is pressed.
125 */
126void
127SimulatorSettingsDialog::onOK(wxCommandEvent&) {
128
129 TransferDataFromWindow();
130
131 std::string command;
132 std::string historyFile =
133 WxConversion::toString(historyFile_.Trim(false).Trim(true));
134
135 if (simulator_.executionTracing() != execTrace_) {
136 command += ProximConstants::SCL_SET + " ";
137 command += ProximConstants::SCL_SETTING_EXEC_TRACE + " ";
138 command += Conversion::toString(execTrace_);
139 command += ProximConstants::SCL_DELIM;
140 }
141
142 if (simulator_.busTracing() != busTrace_) {
143 command += ProximConstants::SCL_SET + " ";
144 command += ProximConstants::SCL_SETTING_BUS_TRACE + " ";
145 command += Conversion::toString(busTrace_);
146 command += ProximConstants::SCL_DELIM;
147 }
148
149 if (simulator_.procedureTransferTracing() != ptTrace_) {
150 command += ProximConstants::SCL_SET + " ";
151 command += ProximConstants::SCL_SETTING_PROCEDURE_TRANSFER_TRACE + " ";
152 command += Conversion::toString(ptTrace_);
153 command += ProximConstants::SCL_DELIM;
154 }
155
156 if (simulator_.rfAccessTracing() != rfTrace_) {
157 command += ProximConstants::SCL_SET + " ";
158 command += ProximConstants::SCL_SETTING_RF_TRACE + " ";
159 command += Conversion::toString(rfTrace_);
160 command += ProximConstants::SCL_DELIM;
161 }
162
163 if (simulator_.fuResourceConflictDetection() != fuConflictDetection_) {
164 command += ProximConstants::SCL_SET + " ";
165 command += ProximConstants::SCL_SETTING_FU_CONFLICT_DETECTION + " ";
166 command += Conversion::toString(fuConflictDetection_);
167 command += ProximConstants::SCL_DELIM;
168 }
169
170 if (lineReader_.saveInputHistoryToFile() != historySave_) {
171 command += ProximConstants::SCL_SET + " ";
172 command += ProximConstants::SCL_SETTING_HISTORY_SAVE + " ";
173 command += Conversion::toString(historySave_);
174 command += ProximConstants::SCL_DELIM;
175 }
176
177 if (lineReader_.inputHistoryMaxLength() != (size_t)historySize_) {
178 command += ProximConstants::SCL_SET + " ";
179 command += ProximConstants::SCL_SETTING_HISTORY_SIZE + " ";
180 command += Conversion::toString(historySize_);
181 command += ProximConstants::SCL_DELIM;
182
183 }
184
185 if (lineReader_.inputHistoryFilename() != historyFile) {
186 command += ProximConstants::SCL_SET + " ";
187 command += ProximConstants::SCL_SETTING_HISTORY_FILE + " ";
188 command += historyFile;
189 command += ProximConstants::SCL_DELIM;
190 }
191
192 if (simulator_.profileDataSaving() != profileDataSaving_) {
193 command += ProximConstants::SCL_SET + " ";
194 command += ProximConstants::SCL_SETTING_PROFILE_DATA_SAVING + " ";
195 command += Conversion::toString(profileDataSaving_);
196 command += ProximConstants::SCL_DELIM;
197 }
198
199 if (simulator_.utilizationDataSaving() != utilizationDataSaving_) {
200 command += ProximConstants::SCL_SET + " ";
201 command += ProximConstants::SCL_SETTING_UTILIZATION_DATA_SAVING + " ";
202 command += Conversion::toString(utilizationDataSaving_);
203 command += ProximConstants::SCL_DELIM;
204 }
205
206 if (simulator_.nextInstructionPrinting() != nextInstructionPrinting_) {
207 command += ProximConstants::SCL_SET + " ";
208 command += ProximConstants::SCL_SETTING_NEXT_INSTRUCTION_PRINTING + " ";
209 command += Conversion::toString(nextInstructionPrinting_);
210 command += ProximConstants::SCL_DELIM;
211 }
212
213 if (simulator_.simulationTimeStatistics() != simulationTimeStatistics_) {
214 command += ProximConstants::SCL_SET + " ";
215 command += ProximConstants::SCL_SETTING_SIMULATION_TIME_STATISTICS + " ";
216 command += Conversion::toString(simulationTimeStatistics_);
217 command += ProximConstants::SCL_DELIM;
218 }
219
220 lineReader_.input(command);
221
222 EndModal(wxID_OK);
223}
