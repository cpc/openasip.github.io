69SectionIdReplacer::tryToReplace() {

71 SectionKey key =
72 SafePointer::sectionKeyFor(reference());
73
74 stream().setWritePosition(streamPosition());
75 stream().writeHalfWord(key.sectionId());

77
78 } catch (KeyNotFound& e) {
79 stream().writeHalfWord(0);
