41/**
42 * Constructor.
43 *
44 * @param subject The SimulationController which is observed.
45 * @param traceDB The Execution Trace Database instance in which the
46 * trace is stored. Expects that the database is open
47 * for writing.
48 */
49ExecutionTracker::ExecutionTracker(
50 TTASimulationController& subject,
51 ExecutionTrace& traceDB) :
52 Listener(), subject_(subject), traceDB_(traceDB) {
53 subject.frontend().eventHandler().registerListener(
54 SimulationEventHandler::SE_CYCLE_END, this);
55}
56
57/**
58 * Destructor.
59 */
60ExecutionTracker::~ExecutionTracker() {
61 subject_.frontend().eventHandler().unregisterListener(
62 SimulationEventHandler::SE_CYCLE_END, this);
63}
64
65/**
66 * Stores instruction execution data in the trace database.
67 *
68 * If any error happens while writing the data, aborts program with
69 * an error message.
70 */
71void
72ExecutionTracker::handleEvent() {
73 try {
74 traceDB_.addInstructionExecution(
75 subject_.clockCount(),
76 subject_.lastExecutedInstruction());
77 } catch (const Exception& e) {
78 debugLog("Error while writing TraceDB: " + e.errorMessage());
79 }
80}
