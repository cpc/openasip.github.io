52/**
53 * Returns the Command corresponding to the type of the Request.
54 *
55 * @param request Request to be handled.
56 * @return NULL if the Request cannot be handled.
57 */
58ComponentCommand*
59ProDeRFPortEditPolicy::getCommand(Request* request) {
60 Request::RequestType type = request->type();
61 if (type == Request::MODIFY_REQUEST) {
62 return new ModifyRFPortCmd(host_);
63 }
64 return NULL;
65}
