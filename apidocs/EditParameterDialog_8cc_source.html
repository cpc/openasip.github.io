97/**
98 * Sets texts for widgets.
99 */
100void
101EditParameterDialog::setTexts() {
102 ProDeTextGenerator* prodeTexts = ProDeTextGenerator::instance();
103
104 // Dialog title
105 format fmt = prodeTexts->text(
106 ProDeTextGenerator::TXT_EDIT_PARAMETER_TITLE);
107 SetTitle(WxConversion::toWxString(fmt.str()));
108
109 paramName_->SetLabel(WxConversion::toWxString(parameter_->name()));
110 paramType_->SetLabel(WxConversion::toWxString(parameter_->typeAsString()));
111 paramValue_->Clear();
112 paramValue_->AppendText(WxConversion::toWxString(parameter_->value()));
113}
