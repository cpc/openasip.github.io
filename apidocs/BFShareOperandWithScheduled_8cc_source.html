48bool BFShareOperandWithScheduled::operator()() {
49 if (removed_.move().destination().operationIndex() !=
50 shared_.move().destination().operationIndex()) {
51#ifdef DEBUG_BUBBLEFISH_SCHEDULER
52 std::cerr << "\t\t\t\t\tDestination operand index mismatch"
53 << std::endl;
54#endif
55 return false;
56 }
57
58 originalRemovedBus_ = &removed_.move().bus();
59 originalCycle_ = removed_.cycle();
60
61 if (removed_.move().source().isImmediateRegister()) {
62 oldImmWriteCycle_ = rm().immediateWriteCycle(removed_);
63 immRegIndex_ = removed_.move().source().index();
64 immu_ = &removed_.move().source().immediateUnit();
65 }
66
67 MoveNode* removedPrologMN = ii() ?
68 duplicator().getMoveNode(removed_) :
69 NULL;
70 if (removedPrologMN != NULL) {
71 prologBus_ = &removedPrologMN->move().bus();
72
73 if (removedPrologMN->move().source().isImmediateRegister()) {
74 prologImmWriteCycle_ =
75 prologRM()->immediateWriteCycle(*removedPrologMN);
76 }
77
78#ifdef DEBUG_BUBBLEFISH_SCHEDULER
79 std::cerr << "\t\t\t\tShareOpWithSched Saving prolog bus: "
80 << prologBus_->name() << std::endl;
81 std::cerr << "\t\t\t\t\tProlog move: " << removedPrologMN->toString()
82 << std::endl;
83#endif
84 }
85
86 unassign(removed_, false);
87 if (BFShareOperand::operator()()) {
88 ddg().dropNode(removed_);
89 sched_.nodeAndCopyKilled(removed_);
90#ifdef DEBUG_BUBBLEFISH_SCHEDULER
91 std::cerr << "\t\t\t\t\tShared operand ok!" << std::endl;
92#endif
93 return true;
94 } else {
95 sched_.nodeResurrected(removed_);
96 ddg().restoreNodeFromParent(removed_);
97 assign(originalCycle_, removed_, originalRemovedBus_);
98#ifdef DEBUG_BUBBLEFISH_SCHEDULER
99 std::cerr << "\t\t\t\t\tShare operand failed!" << std::endl;
100#endif
101 return false;
102 }
103}
