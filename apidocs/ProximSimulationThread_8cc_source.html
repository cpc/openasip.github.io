112 simulation_->registerListener(
113 TracedSimulatorFrontend::SIMULATOR_RESET, this);

115 runtimeErrorHandler_ = new ProximRuntimeErrorHandler(*simulation_, *gui_);

119/**
120 * Entry point for the simulation thread execution.
121 *
122 * The thread waits commands from the line reader, and executes
123 * them using the simulator interpreter. Commands are executed
124 * until the thread is terminated.
125 */
126void*
127ProximSimulationThread::Entry() {
128
129 assert(gui_ != NULL);
130
131 while (!TestDestroy()) {
132
133 std::string command = "";
134
135 command = lineReader_->readLine();
136 command = StringTools::trim(command);
137
138 if (command == "") {
139 continue;
140 }
141
142 if (command == "quit") {
143 // Post an event which tells the gui thread to delete this thread.
144 SimulatorEvent eventCmd(
145 SimulatorEvent::EVT_SIMULATOR_TERMINATED);
146 wxPostEvent(gui_, eventCmd);
147
148 // Wait for the gui thread to call Delete() for this thread.
149 while (!TestDestroy()) {
150 Sleep(100);
151 }
152 return 0;
153 }
154
155 // Inform GUI about the command being executed.
156 SimulatorEvent eventCmd(
157 SimulatorEvent::EVT_SIMULATOR_COMMAND, command);
158 wxPostEvent(gui_, eventCmd);
159
160 interpreter_->interpret(command);
161
162 // Inform GUI about the command execution being done.
163 SimulatorEvent eventCmdDone(
164 SimulatorEvent::EVT_SIMULATOR_COMMAND_DONE, command);
165 wxPostEvent(gui_, eventCmdDone);
166
167 // If the command resulted in error, inform the GUI about the error.
168 if (interpreter_->error()) {
169 SimulatorEvent eventError(
170 SimulatorEvent::EVT_SIMULATOR_ERROR,
171 interpreter_->result());
172 wxPostEvent(gui_, eventError);
173 continue;
174 }
175
176 // Inform GUI about the simulator interpreter result.
177 if (interpreter_->result().size() > 0) {
178 SimulatorEvent event(
179 SimulatorEvent::EVT_SIMULATOR_OUTPUT,
180 (interpreter_->result() + "\n"));
181 wxPostEvent(gui_, event);
182 }
183 }
184 return 0;
185}

250/**
251 * Handles events from the simulator backend.
252 *
253 * The events are passed to the GUI as SimulatorEvents, which is a class
254 * derived from the wxEvent class. This way responsibility of the event
255 * handling can be passed to the GUI thread.
256 */
257void
258ProximSimulationThread::handleEvent(int event) {
259
260 switch (event) {
261 case TracedSimulatorFrontend::SIMULATOR_LOADING_MACHINE: {
262 SimulatorEvent simEvent(SimulatorEvent::EVT_SIMULATOR_LOADING_MACHINE);
263 wxPostEvent(gui_, simEvent);
264 return;
265 }
266 case TracedSimulatorFrontend::SIMULATOR_MACHINE_LOADED: {
267 SimulatorEvent simEvent(SimulatorEvent::EVT_SIMULATOR_MACHINE_LOADED);
268 wxPostEvent(gui_, simEvent);
269 return;
270 }
271 case TracedSimulatorFrontend::SIMULATOR_LOADING_PROGRAM: {
272 SimulatorEvent simEvent(SimulatorEvent::EVT_SIMULATOR_LOADING_PROGRAM);
273 wxPostEvent(gui_, simEvent);
274 return;
275 }
276 case TracedSimulatorFrontend::SIMULATOR_PROGRAM_LOADED: {
277 SimulatorEvent simEvent(SimulatorEvent::EVT_SIMULATOR_PROGRAM_LOADED);
278 wxPostEvent(gui_, simEvent);
279 return;
280 }
281 case TracedSimulatorFrontend::SIMULATOR_START: {
282 SimulatorEvent simEvent(SimulatorEvent::EVT_SIMULATOR_START);
283 wxPostEvent(gui_, simEvent);
284 return;
285 }
286 case TracedSimulatorFrontend::SIMULATOR_STOP: {
287 SimulatorEvent simEvent(SimulatorEvent::EVT_SIMULATOR_STOP);
288 wxPostEvent(gui_, simEvent);
289 return;
290 }
291 case TracedSimulatorFrontend::SIMULATOR_RUN: {
292 SimulatorEvent simEvent(SimulatorEvent::EVT_SIMULATOR_RUN);
293 wxPostEvent(gui_, simEvent);
294 return;
295 }
296 case TracedSimulatorFrontend::SIMULATOR_RESET: {
297 gui_->reset();
298 return;
299 }
300 };
301}
