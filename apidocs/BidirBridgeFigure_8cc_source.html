59/**
60 * Lays out the two child bridges.
61 *
62 * @param dc Device context.
63 */
64void
65BidirBridgeFigure::layoutChildren(wxDC* dc) {
66
67 assert(children_.size() == 2);
68
69 dynamic_cast<BridgeFigure*>(
70 children_[0])->setDirection(BridgeFigure::DIR_LEFT);
71
72 children_[0]->setX(location_.x);
73 children_[0]->setY(location_.y);
74 children_[0]->setWidth(size_.GetWidth());
75 children_[0]->layout(dc);
76
77 dynamic_cast<BridgeFigure*>(
78 children_[1])->setDirection(BridgeFigure::DIR_RIGHT);
79
80 children_[1]->setX(location_.x);
81 children_[1]->setY(
82 location_.y + MachineCanvasLayoutConstraints::BRIDGE_HEIGHT);
83
84 children_[1]->setWidth(size_.GetWidth());
85 children_[1]->layout(dc);
86}
