56/**
57 * A templated class for classes storing inter-pass data of any type.
58 *
59 * Inherits the interface of the given class and the InterPassDatum, thus
60 * acts like the given class, but can be stored as inter-pass data. Should
61 * make creating new InterPassDatum types easier.
62 */
63template <typename T>
64class SimpleInterPassDatum : public InterPassDatum, public T {
65public:
66 SimpleInterPassDatum() : InterPassDatum(), T() {}
67 virtual ~SimpleInterPassDatum() {};
68};
